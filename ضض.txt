    <div class="question" id="question">جارٍ تحميل السؤال...</div>
    <div class="choices" id="choices"></div>
    <div class="result" id="result"></div>
  </div>

  <script>
    const questions = [
      {
        question: "ما هي عاصمة السعودية؟",
        choices: ["الأردن", "عمان", "الرياض", "جازان"],
        answer: "الرياض"
      },
      {
        question: "ما هو أطول نهر في العالم؟",
        choices: ["الأمازون", "الفرات", "النيل", "الدانوب"],
        answer: "النيل"
      },
      {
        question: "كم عدد الكواكب في المجموعة الشمسية؟",
        choices: ["7", "8", "9", "6"],
        answer: "8"
      },
      {
        question: "من هو مؤسس شركة مايكروسوفت؟",
        choices: ["إيلون ماسك", "بيل غيتس", "ستيف جوبز", "مارك زوكربيرغ"],
        answer: "بيل غيتس"
      },
      {
        question: "ما هي عاصمة مصر؟",
        choices: ["الخرطوم", "القاهرة", "الرباط", "دمشق"],
        answer: "القاهرة"
      }
    ];

    let currentQuestion = 0;
    let score = 0;
    let timeLeft = 10;
    let timerInterval;

    const questionElement = document.getElementById("question");
    const choicesElement = document.getElementById("choices");
    const timerElement = document.getElementById("timer");
    const resultElement = document.getElementById("result");

    function showQuestion() {
      clearInterval(timerInterval);
      timeLeft = 10;
      timerElement.textContent = "الوقت: 10 ثواني";

      const q = questions[currentQuestion];
      questionElement.textContent = q.question;
      choicesElement.innerHTML = "";

      q.choices.forEach(choice => {
        const btn = document.createElement("button");
        btn.textContent = choice;
        btn.onclick = () => selectAnswer(choice);
        choicesElement.appendChild(btn);
      });

      timerInterval = setInterval(() => {
        timeLeft--;
        timerElement.textContent = "الوقت: " + timeLeft + " ثواني";
        if (timeLeft <= 0) {
          clearInterval(timerInterval);
          selectAnswer(null); // يعتبر لم يجاوب
        }
      }, 1000);
    }

    function selectAnswer(selected) {
      clearInterval(timerInterval);

      const correct = questions[currentQuestion].answer;
      if (selected === correct) {
        score++;
      }

      currentQuestion++;
      if (currentQuestion < questions.length) {
        showQuestion();
      } else {
        showResult();
      }
    }

    function showResult() {
      questionElement.textContent = "انتهى الاختبار!";
      choicesElement.innerHTML = "";
      timerElement.textContent = "";
      resultElement.textContent = `درجتك: ${score} من ${questions.length}`;
    }

    // ابدأ أول سؤال
    showQuestion();
  </script>
